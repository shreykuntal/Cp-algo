vector<int> pd(int n) 
{ 
	vector<int> ans;
	// Note that this loop runs till square root 
	for (int i=1; i<=sqrtl(n); i++) 
	{ 
		if (n%i == 0) 
		{ 
			// If divisors are equal, print only one 
			if (n/i == i) 
				ans.push_back(i); 
 
			else // Otherwise print both 
				{ans.push_back(i); ans.push_back(n/i);} 
		} 
	} 
	sort(ans.begin(), ans.end());
	return ans;
}
